UPDATED PRICING CONSTANTS const PLAN_RATES = { annual: 1116.00, quarterly: 329.00, // Updated from original rate family_add_on: 1000.00, // Paid in full, no installments add_on_visit: 129.00, add_on_massage: 50.00 }; // 2. AUTOMATED STATUS LOGIC (For Addison & Abrams Tabs) function determinePatientStatus(patient, appointments) { const hasMedicalVisitThisQuarter = checkUtilization(patient.mrn, 'medical'); const hasMassageThisQuarter = checkUtilization(patient.mrn, 'massage'); // If a visit already exists in the 90-day bucket, label as "Add-On" if (appointments.type === 'KEP_EST' && hasMedicalVisitThisQuarter) { return { label: 'Add-On', charge: PLAN_RATES.add_on_visit }; } if (appointments.type === 'KEPmasg' && hasMassageThisQuarter) { return { label: 'Add-On', charge: PLAN_RATES.add_on_massage }; } // Otherwise, it remains a standard Plan Visit return { label: 'Active', charge: 0.00 }; } // 3. EXPIRING TAB LOGIC (Assembly Line 7-Day Gate) function generateExpiringReport(members) { return members.filter(m => isWithin7Days(m.nextChargeDate)).map(m => { let amount = PLAN_RATES.annual; if (m.plan === 'Quarterly') amount = PLAN_RATES.quarterly; if (m.plan === 'Family') amount = PLAN_RATES.family_add_on; return { ...m, amountToCollect: amount }; }); }
